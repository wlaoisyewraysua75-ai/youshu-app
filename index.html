// === 核心逻辑：本地缓存 + 异步刷新 === const CACHE_KEY = 'youshu_data_cache'; let currentActiveRowId = null; // 用于记录当前打开的物品的行号 let currentActiveItem = null; // 记录当前打开的物品完整数据 async function loadData() { const cachedData = localStorage.getItem(CACHE_KEY); if (cachedData) { const data = JSON.parse(cachedData); renderItems(data); calculateOverview(data); } else { itemListContainer.innerHTML = '
首次加载，正在从云端同步...
'; } try { const response = await fetch(API_URL); const data = await response.json(); localStorage.setItem(CACHE_KEY, JSON.stringify(data)); renderItems(data); calculateOverview(data); } catch (error) { if (!cachedData) itemListContainer.innerHTML = '
加载失败，请检查网络。
'; } } const detailDrawer = document.getElementById('detail-drawer'); const drawerClose = document.getElementById('drawer-close'); drawerClose.addEventListener('click', () => detailDrawer.style.display = 'none'); function renderItems(data) { itemListContainer.innerHTML = ''; data.forEach(item => { const days = calculateDays(item['购入日期']); const price = parseFloat(item['购入价格']) || 0; const dailyCost = price / days; let icon = '📦'; if (item['一级分类'] === '交通出行') icon = '🛵'; else if (item['一级分类'] === '电子设备') icon = '💻'; else if (item['一级分类'] === '兴趣爱好') icon = '🃏'; const card = document.createElement('div'); card.className = 'item-card'; card.innerHTML = `
${item['当前状态']}
${icon}
${item['物品名称']}
¥${formatMoney(price)} | 已使用 ${days} 天
¥${formatMoney(dailyCost)}/天
`; card.addEventListener('click', () => { // 点击卡片时，记录下行号和数据 currentActiveRowId = item._rowIndex; currentActiveItem = item; document.getElementById('drawer-title').innerText = item['物品名称']; document.getElementById('drawer-tags').innerText = `${item['一级分类']} - ${item['二级分类'] || '无标签'}`; // 修复日期显示可能带有时区T的格式 document.getElementById('drawer-date').innerText = item['购入日期'].split('T')[0]; document.getElementById('drawer-notes').innerText = item['备注'] || '暂无备注...'; detailDrawer.style.display = 'flex'; }); itemListContainer.appendChild(card); }); } // === 新增/编辑/删除 的表单逻辑 === const addModal = document.getElementById('add-modal'); const btnAdd = document.getElementById('add-btn'); const btnCancel = document.getElementById('btn-cancel'); const btnSubmit = document.getElementById('btn-submit'); const inputDate = document.getElementById('input-date'); // 辅助函数：清空表单 const clearForm = () => { ['input-name', 'input-price', 'input-category2', 'input-notes'].forEach(id => document.getElementById(id).value = ''); document.getElementById('input-category1').selectedIndex = 0; document.getElementById('input-status').selectedIndex = 0; }; // 点击加号：进入“新增”模式 btnAdd.addEventListener('click', () => { currentActiveRowId = null; // 清除行号，代表这是新增操作 document.getElementById('modal-title').innerText = '录入新资产'; clearForm(); inputDate.valueAsDate = new Date(); addModal.style.display = 'flex'; }); btnCancel.addEventListener('click', () => addModal.style.display = 'none'); // 抽屉内点击编辑：进入“修改”模式，回填数据 document.getElementById('btn-edit').addEventListener('click', () => { detailDrawer.style.display = 'none'; document.getElementById('modal-title').innerText = '修改资产信息'; document.getElementById('input-name').value = currentActiveItem['物品名称']; document.getElementById('input-price').value = currentActiveItem['购入价格']; // 格式化日期回填 const rawDate = currentActiveItem['购入日期'].replace(/\//g, '-').split('T')[0]; document.getElementById('input-date').value = rawDate; document.getElementById('input-category1').value = currentActiveItem['一级分类'] || '电子设备'; document.getElementById('input-status').value = currentActiveItem['当前状态'] || '服役中'; document.getElementById('input-category2').value = currentActiveItem['二级分类'] || ''; document.getElementById('input-notes').value = currentActiveItem['备注'] || ''; addModal.style.display = 'flex'; }); // 抽屉内点击删除 document.getElementById('btn-delete').addEventListener('click', async () => { if (!confirm(`确定要删除 [${currentActiveItem['物品名称']}] 吗？此操作无法撤销。`)) return; detailDrawer.style.display = 'none'; const payload = { action: "delete", rowId: currentActiveRowId }; // 先在本地缓存中乐观删除，实现秒级响应 const oldCache = JSON.parse(localStorage.getItem(CACHE_KEY) || '[]'); const newCache = oldCache.filter(item => item._rowIndex !== currentActiveRowId); localStorage.setItem(CACHE_KEY, JSON.stringify(newCache)); renderItems(newCache); calculateOverview(newCache); // 发送给云端 fetch(API_URL, { method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8' }, body: JSON.stringify(payload) }).then(() => loadData()); // 异步对齐云端数据 }); // 统一处理保存（包含新增和修改） btnSubmit.addEventListener('click', async () => { const name = document.getElementById('input-name').value; const price = document.getElementById('input-price').value; const date = inputDate.value; const cat1 = document.getElementById('input-category1').value; const status = document.getElementById('input-status').value; const cat2 = document.getElementById('input-category2').value; const notes = document.getElementById('input-notes').value; if (!name || !price || !date) { alert('名称、价格和日期是必填的哦！'); return; } const originalText = btnSubmit.innerText; btnSubmit.innerText = '正在同步...'; btnSubmit.disabled = true; const payloadData = { "物品名称": name, "购入日期": date, "购入价格": price, "当前状态": status, "一级分类": cat1, "二级分类": cat2, "备注": notes }; // 判断是新增还是修改 const payload = { action: currentActiveRowId ? "edit" : "add", rowId: currentActiveRowId, data: payloadData }; try { const response = await fetch(API_URL, { method: 'POST', headers: { 'Content-Type': 'text/plain;charset=utf-8' }, body: JSON.stringify(payload) }); const result = await response.json(); if (result.status === 'success') { addModal.style.display = 'none'; btnSubmit.innerText = originalText; btnSubmit.disabled = false; loadData(); // 重新拉取并更新缓存 } } catch (error) { alert('写入失败，请检查网络设置。'); btnSubmit.innerText = originalText; btnSubmit.disabled = false; } });
